```cs
using System;
using System.Collections.Generic;

namespace exercicio1
{
    class Program
    {
        
        private static int[] arraynumeros = new int[3]; //aqui é a variável onde vou guardar o input do user (após conversão)
        private static string[] userInput = new string[3]; //aqui é a variavel array onde vai buscar o input do user
        private static int userchoiceint; //aqui é a variável onde converto pa int quantos numeros o user quer meter

        //aqui é onde começa o programa e chamo os métodos
        private static void Main(string[] args) {
            AddArray(); //aqui peço os números ao user
            ConvertArray(); //aqui converto os inputs do user        
            Options(); //aqui dou as opções ao user e processo o que ele decidir
        }

        //aqui peço os números ao user
        private static void AddArray() {
            //quantos números quer colocar?
            Console.WriteLine("\n Quantos números quer colocar? \n");
            string userchoice = Console.ReadLine();
            userchoiceint = Convert.ToInt32(userchoice);
            //
            //faço resize nas arrays de acordo com o número que o user escolher
            Array.Resize<string>(ref userInput, userchoiceint);
            Array.Resize<int>(ref arraynumeros, userchoiceint);
            //
            //pedir os números
            int i = 0;
            for(i = 0; i < userInput.Length; i++) {
                Console.WriteLine("\n Escreve um número! \n");
                userInput[i] = Console.ReadLine();
            }
            //
        }

        //aqui converto os inputs do user
        private static void ConvertArray() {
            arraynumeros = Array.ConvertAll(userInput, int.Parse);
            foreach(var item in arraynumeros) {
                Console.WriteLine(item.ToString());
            }
        }

        //aqui dou as opções ao user e processo o que ele decidir
        private static void Options() {
            string userChoice;
            int i = 0;
            for (i = 0; i < 6; i--) {
                Console.WriteLine("\n O que queres fazer? \n");
                Console.WriteLine("\n 1 - Obter média dos valores no array");
                Console.WriteLine("\n 2 - Obter mediana dos valores no array");
                Console.WriteLine("\n 3 - Obter moda dos valores no array");
                Console.WriteLine("\n 4 - Obter máximo dos valores no array");
                Console.WriteLine("\n 5 - Obter mínimo dos valores no_array");
                Console.WriteLine("\n 6 - Sair \n");
                userChoice = Console.ReadLine();
                if(userChoice == "1") {
                    Avg(); //método para calcular a média
                }
                if (userChoice == "2") {
                    Median(); //método para calcular a mediana
                }
                if (userChoice == "3") {
                    Mode(); //método para calcular a moda
                }
                if (userChoice == "4") {
                    Max(); //método para calcular o máximo
                }
                if (userChoice == "5") {
                    Min(); //método para calcular o mínimo
                }
                if (userChoice == "6") {
                    Exit(); //método para sair do programa
                }
            }
        }

        //método para calcular a média
        private static void Avg() {
            Console.WriteLine("\n Média \n");
            int sum = 0;
            //aqui abaixo vou buscar a soma de todo o conteúdo da array
            foreach(int number in arraynumeros) {
                sum = sum + number;
            }
            //
            Console.WriteLine(sum / userchoiceint); //aqui faço a divisão da soma com o tamanho da array
        }

        //método para calcular a mediana
        private static void Median() {
            Console.WriteLine("\n Mediana \n");
            //na array order vou guardar o conteúdo da arraynumber pa ordenar
            int[] order = new int[userchoiceint]; 
            //
            int counter = 0;
            //neste foreach passo os números do arraynumeros para a array order para depois ordenar
            foreach(int number in arraynumeros) {
                order[counter] = number;
                counter++;
            }
            //
            Array.Sort(order); //para ordenar os números do array por ordem crescente

            if (order.Length % 2 == 0) { //é par?
                Console.WriteLine("Par");
                //isto aqui abaixo é para perceber quais são os 2 números no meio
                int midhigh = order[order.Length / 2];
                int midlow = order[order.Length / 2] - 1;
                //
                //aqui abaixo faço a média desses 2 números do meio
                int result = (midhigh + midlow) / userchoiceint;
                Console.WriteLine(result);
            }
            else { //é impar?
                Console.WriteLine("Impar"); 
                int mid = order[order.Length / 2]; //aqui busco o número do meio
                Console.WriteLine(mid); //mostra o número do meio (numero de elementos impar)
            }
        }

        //método para calcular a moda
        //vou refazer este em breve com a minha cabeça
        private static void Mode() {
            Console.WriteLine("\n Moda \n");
            Dictionary<int, int> counts = new Dictionary<int, int>();
            foreach(int a in arraynumeros) {
                if(counts.ContainsKey(a)) {
                    counts[a] = counts[a] + 1;
                } else {
                    counts[a] = 1;
                }
            }

            int result = int.MinValue;
            int max = int.MinValue;
            foreach(int key in counts.Keys) {
                if(counts[key] > max) {
                    max = counts[key];
                    result = key;
                }
            }

            Console.WriteLine("A moda é = " + result);
        }
        //

        //método para calcular o máximo
        //a ideia é ordenar os valores por ordem crescente e depois
        //apanhar o index mais alto (que é o número mais alto)
        private static void Max() {
            Console.WriteLine("\n Máximo \n");
            //na array order vou guardar o conteúdo da arraynumeros pa ordenar
            int[] order = new int[userchoiceint]; 
            int counter = 0;
            //
            //neste foreach passo os números do arraynumeros para a array order para depois ordenar
            foreach (int number in arraynumeros) {
                order[counter] = number;
                counter++;
            }
            //
            Array.Sort(order); //para ordenar os números do array por ordem crescente
            //aqui abaixo printo o número que está no topo da array
            Console.WriteLine(order[userchoiceint -1]); 
            //
        }

        //método para calcular o mínimo
        //a ideia é ordenar os valores por ordem crescente e depois
        //apanhar o index mais abaixo (que é o número mais baixo)
        private static void Min() {
            Console.WriteLine("\n Mínimo \n");
            //na array order vou guardar o conteúdo da arraynumber pa ordenar
            int[] order = new int[userchoiceint];
            int counter = 0;
            //
            //neste foreach passo os números do arraynumeros para a array order para depois ordenar
            foreach (int number in arraynumeros) {
                order[counter] = number;
                counter++;
            }
            //
            Array.Sort(order); //para ordenar os números do array por ordem crescente
            //aqui abaixo printo o número que está mais abaixo na array
            Console.WriteLine(order[userchoiceint - userchoiceint]);
            //
        }

        //método para sair do programa
       private static void Exit() {
            Console.WriteLine("\n Sair \n");
            Environment.Exit(0);
        }
    }
}
```
